# Custom props

<CodeGroup>
  ```JavaScript
	import { Turnstile } from '@marsidev/react-turnstile'

	export default function Widget() {
		return (
			<Turnstile
				siteKey='{{ sitekey }}'
				className='fixed bottom-4 right-4'
				options={{
					action: 'submit-form',
					theme: 'light',
					size: 'compact',
					language: 'fr',
				}}
				scriptOptions={{
					appendTo: 'body'
				}}
			/>
		)
	}
	```
